        function parseFile(content) {

            items = [];

            

            if (currentMode === 'facts') {

                const facts = content.split('---').filter(f => f.trim());

                items = facts.map(fact => ({

                    type: 'fact',

                    content: fact.trim()

                }));

            } else {

                const quizzes = content.split('---\n\n').filter(q => q.trim());

                quizzes.forEach(quiz => {

                    const lines = quiz.split('\n').filter(l => l.trim());

                    if (lines.length > 0) {

                        const quizObj = {

                            type: 'quiz',

                            raw: quiz.trim(),

                            title: '',

                            quizType: '',

                            difficulty: '',

                            explanation: '',

                            question: '',

                            options: []

                        };


                        let inMeta = false;

                        let inOptions = false;


                        for (let line of lines) {

                            if (line.startsWith('### Quiz:')) {

                                quizObj.title = line.replace('### Quiz:', '').trim();

                                inMeta = true;

                            } else if (inMeta && line.startsWith('- type:')) {

                                quizObj.quizType = line.replace('- type:', '').trim();

                            } else if (inMeta && line.startsWith('- difficulty:')) {

                                quizObj.difficulty = line.replace('- difficulty:', '').trim();

                            } else if (inMeta && line.startsWith('- explanation:')) {

                                quizObj.explanation = line.replace('- explanation:', '').trim();

                                inMeta = false;

                            } else if (line.startsWith('- ') && !inMeta) {

                                inOptions = true;

                                quizObj.options.push(line);

                            } else if (!inMeta && !inOptions && line.trim() && !line.startsWith('###')) {

                                quizObj.question = line.trim();

                            }

                        }


                        items.push(quizObj);

                    }

                });

            }


            currentIndex = 0;

            approved = [];

            rejected = [];

            updateStats();

            renderCard();

        }

        function addSwipeListeners(card) {

            card.addEventListener('mousedown', startDrag);

            card.addEventListener('touchstart', startDrag);

            document.addEventListener('mousemove', drag);

            document.addEventListener('touchmove', drag);

            document.addEventListener('mouseup', endDrag);

            document.addEventListener('touchend', endDrag);

        }


        function startDrag(e) {

            if (isEditing) return;

            isDragging = true;

            startX = e.type.includes('mouse') ? e.clientX : e.touches[0].clientX;

            currentX = startX;

        }


        function drag(e) {

            if (!isDragging || isEditing) return;

            

            currentX = e.type.includes('mouse') ? e.clientX : e.touches[0].clientX;

            const diff = currentX - startX;

            const card = cardContainer.querySelector('.card');

            

            if (card) {

                card.style.transform = `translateX(${diff}px) rotate(${diff * 0.05}deg)`;

                

                if (diff < -50) {

                    card.classList.add('swiping-left');

                    card.classList.remove('swiping-right');

                    leftIndicator.classList.add('visible');

                    rightIndicator.classList.remove('visible');

                } else if (diff > 50) {

                    card.classList.add('swiping-right');

                    card.classList.remove('swiping-left');

                    rightIndicator.classList.add('visible');

                    leftIndicator.classList.remove('visible');

                } else {

                    card.classList.remove('swiping-left', 'swiping-right');

                    leftIndicator.classList.remove('visible');

                    rightIndicator.classList.remove('visible');

                }

            }

        }


        function endDrag() {

            if (!isDragging || isEditing) return;

            isDragging = false;

            

            const diff = currentX - startX;

            const card = cardContainer.querySelector('.card');

            

            leftIndicator.classList.remove('visible');

            rightIndicator.classList.remove('visible');

            

            if (Math.abs(diff) > 100) {

                if (diff < 0) {

                    reject();

                } else {

                    approve();

                }

            } else if (card) {

                card.style.transform = '';

                card.classList.remove('swiping-left', 'swiping-right');

            }

        }


        function approve() {

            approved.push(items[currentIndex]);

            nextCard();

        }


        function reject() {

            rejected.push(items[currentIndex]);

            nextCard();

        }


        function nextCard() {

            currentIndex++;

            updateStats();

            renderCard();

        }


        function updateStats() {

            document.getElementById('approvedCount').textContent = approved.length;

            document.getElementById('rejectedCount').textContent = rejected.length;

            document.getElementById('remainingCount').textContent = Math.max(0, items.length - currentIndex);

            

            downloadBtn.disabled = approved.length === 0;

        }


        function editItem() {

            isEditing = true;

            const card = cardContainer.querySelector('.card');

            const item = items[currentIndex];

            

            const currentContent = item.type === 'fact' ? item.content : item.raw;

            

            card.innerHTML = `

                <div class="card-content">

                    <textarea class="edit-textarea" id="editTextarea">${currentContent}</textarea>

                    <div class="edit-actions">

                        <button class="save-btn" onclick="saveEdit()">Save</button>

                        <button class="cancel-btn" onclick="cancelEdit()">Cancel</button>

                    </div>

                </div>

            `;

        }


        function saveEdit() {

            const textarea = document.getElementById('editTextarea');

            const newContent = textarea.value.trim();

            

            if (items[currentIndex].type === 'fact') {

                items[currentIndex].content = newContent;

            } else {

                items[currentIndex].raw = newContent;

                // Re-parse the quiz

                const lines = newContent.split('\n').filter(l => l.trim());

                // Update quiz fields (simplified)

                items[currentIndex].title = lines.find(l => l.startsWith('### Quiz:'))?.replace('### Quiz:', '').trim() || '';

            }

            

            isEditing = false;

            renderCard();

        }


        function cancelEdit() {

            isEditing = false;

            renderCard();

        }


        approveBtn.addEventListener('click', approve);

        rejectBtn.addEventListener('click', reject);


        downloadBtn.addEventListener('click', () => {

            let content = '';

            

            if (currentMode === 'facts') {

                content = approved.map(item => item.content).join('\n---\n---\n');

            } else {

                content = approved.map(item => item.raw).join('\n---\n\n');

            }

            

            const blob = new Blob([content], { type: 'text/plain' });

            const url = URL.createObjectURL(blob);

            const a = document.createElement('a');

            a.href = url;

            a.download = `approved-${currentMode}-${Date.now()}.txt`;

            a.click();

            URL.revokeObjectURL(url);

        });


        function resetApp() {

            items = [];

            currentIndex = 0;

            approved = [];

            rejected = [];

            updateStats();

            cardContainer.innerHTML = `

                <div class="empty-state">

                    <div class="empty-state-icon">📚</div>

                    <div class="empty-state-text">Upload a file to get started</div>

                </div>

            `;

            fileInput.value = '';

        }
